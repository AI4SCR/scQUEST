[docs]def test_reproducible_model_init():
    m1 = DefaultAE(5, [3, 3])
    m2 = DefaultAE(5, [3, 3])

    M = [m1, m2]
    P = [m.parameters() for m in M]
    for w1, w2 in zip(*P):
        assert (w1 == w2).all()
